# 手柄连接检测功能实现

## 目标
在 `_wait_for_gamepad` 函数中使用 Python pygame 或 Windows API 检测手柄连接状态，并获取手柄信息。

## 技术方案

### 方案 1：使用 pygame（推荐）

```python
import pygame

def get_gamepad_info():
    """获取已连接的手柄信息"""
    pygame.init()
    pygame.joystick.init()

    joystick_count = pygame.joystick.get_count()
    if joystick_count == 0:
        return None

    joystick = pygame.joystick.Joystick(0)
    joystick.init()

    info = {
        'name': joystick.get_name(),
        'axes': joystick.get_numaxes(),
        'buttons': joystick.get_numbuttons(),
        'hats': joystick.get_numhats(),
        'connected': True
    }
    return info
```

## 实施步骤

### 1. 修改 game_automation.py

添加手柄检测相关代码：

```python
def _get_gamepad_info(self) -> dict:
    """获取已连接的手柄信息"""
    try:
        import pygame
        pygame.init()
        pygame.joystick.init()

        joystick_count = pygame.joystick.get_count()
        if joystick_count == 0:
            logger.info("未检测到手柄连接")
            return None

        joystick = pygame.joystick.Joystick(0)
        joystick.init()

        info = {
            'name': joystick.get_name(),
            'axes': joystick.get_numaxes(),
            'buttons': joystick.get_numbuttons(),
            'hats': joystick.get_numhats(),
        }
        logger.info(f"检测到手柄: {info['name']}, 轴数: {info['axes']}, 按钮数: {info['buttons']}")
        return info
    except Exception as e:
        logger.warning(f"获取手柄信息失败: {e}")
        return None

def _wait_for_gamepad(self, timeout: int = 10) -> bool:
    """等待手柄连接"""
    import pygame
    import time

    logger.info(f"等待手柄连接 (超时: {timeout}秒)...")
    start_time = time.time()

    while time.time() - start_time < timeout:
        info = self._get_gamepad_info()
        if info:
            logger.info(f"手柄已连接: {info['name']}")
            return True
        time.sleep(0.5)

    logger.warning("等待手柄连接超时")
    return False
```

## 注意事项

1. pygame 需要初始化 `pygame.joystick.init()` 才能检测手柄
2. 如果手柄数量为 0，表示没有手柄连接
3. Xbox 手柄信息：
   - 轴数：4（左右摇杆各2个轴）
   - 按钮数：17（包含方向键）
